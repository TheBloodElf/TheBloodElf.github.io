获取应用缓存

代码如下，直接拷贝，
//获取单个文件的大小

  • (long long) fileSizeAtPath:(NSString) filePath{
    NSFileManager     manager = [NSFileManager defaultManager];
    if ([manager fileExistsAtPath:filePath]){
    return [[manager attributesOfItemAtPath:filePath error:nil] fileSize];
    }
    return 0;
    }
    //遍历文件夹获得文件夹大小，返回多少M
  • (float ) folderSizeAtPath:(NSString) folderPath{
    NSFileManager     manager = [NSFileManager defaultManager];
    if (![manager fileExistsAtPath:folderPath]) return 0;
    NSEnumerator childFilesEnumerator = [[manager subpathsAtPath:folderPath] objectEnumerator];
    NSString     fileName;
    long long folderSize = 0;
    while ((fileName = [childFilesEnumerator nextObject]) != nil){
    NSString fileAbsolutePath = [folderPath stringByAppendingPathComponent:fileName];
    folderSize += [self fileSizeAtPath:fileAbsolutePath];
    }
    return folderSize/(1024.0    1024.0);
    }
    //得到系统中总的缓存大小
  • (float)getSizeOfAPP
    {
    NSArray paths = NSSearchPathForDirectoriesInDomains(NSLibraryDirectory, NSUserDomainMask, YES);
    NSString     path = [paths lastObject];
    return [self folderSizeAtPath:path];
    }
    使用方法：[self getSizeOfApp],就返回大小了，是m为单位，可以自己改

把手机号输入格式限制成3,4,4位(如:135 5555 5555)

带入如下，思想就是做一个过滤和格式化：

#pragma mark 把电话号码格式变成135 5555 5555的样子
-(BOOL)textField:(UITextField )textField shouldChangeCharactersInRange:(NSRange)range replacementString:(NSString )string {
if(textField.tag == DianHua)
{
NSCharacterSet characterSet = [NSCharacterSet characterSetWithCharactersInString:@”0123456789\b”];
string = [string stringByReplacingOccurrencesOfString:@” “ withString:@””];
if ([string rangeOfCharacterFromSet:[characterSet invertedSet]].location != NSNotFound) {
return NO;
}
NSString text = [textField text];
text = [text stringByReplacingCharactersInRange:range withString:string];
text = [text stringByReplacingOccurrencesOfString:@” “ withString:@””];
if (text.length == 0) {
[textField setText:@””];
}
else if (text.length <= 3) {
NSString str1 = [text substringWithRange:NSMakeRange(0, text.length)];
[textField setText:[NSString stringWithFormat:@”%@”, str1]];
}
else if (text.length > 3 && text.length <= 7) {
NSString str1 = [text substringWithRange:NSMakeRange(0, 3)];
NSString str2 = [text substringWithRange:NSMakeRange(3, text.length - 3)];
[textField setText:[NSString stringWithFormat:@”%@ %@”, str1, str2]];
}
else if (text.length > 7 && text.length <= 11) {
NSString str1 = [text substringWithRange:NSMakeRange(0, 3)];
NSString str2 = [text substringWithRange:NSMakeRange(3, 4)];
NSString str3 = [text substringWithRange:NSMakeRange(7, text.length - 7)];
[textField setText:[NSString stringWithFormat:@”%@ %@ %@”, str1, str2, str3]];
}
return NO;
}
return YES;
}

这个是写给唐巧哥的，同时间发出来给你们。
原文：
” 你好，巧哥！我是很崇拜你的人，我也是一个IOS开发人员，只不过才工作两个月！全是自学的IOS！

这个点子的产生是因为我在等公交的时候（真实的哦，所以更想有这个软件）。有一天晚上我在等公交，我发现公交收车了（当时是9点30），于是我又百度其他路线回家，找了20分钟（因为还是需要避开已经收车的公交）终于找到了。但是发现乘坐时间是一个小时二十分钟，所以我想打UBER（因为还有事情要做），但是发现要19块钱。我还是放弃了，但是这个时候旁边的两位要坐公交的阿姨说“怎么收车了”，我才去问她“你们是要到哪里？”，然后才知道和我的路线一样，所以我们三个就拼车回去了，时间只要20分钟，价格每个人只多了2元！

公司项目流程

这篇文章是写给找工作的人的，因为面试时候会问到这个问题，那么公司的项目流程是什么样的呢？
一：先有点子，点子是谁提供的呢？这个得看公司的制度了，是专人或者所有人集合一起。
二：有了点子不是就可以开始动手了，得先大概的想一想这个方案能不能做出来？这时候需要技术人员和需求人员等相关人一起讨论。
三：如果方案不可行，则另外想或者减少方案中复杂的部分。
四：如果方案可行，那么就可以动手了（除了客户端），比如原型设计师做原型，我们公司用的原型设计工具是Briefscase；服务器的人就可以做接口了（需要和客户端的沟通）；如果项目是有其他部分的，比如微信公众号、网页端等等，那么这些人也可以开始动手了，因为这些往往是客户端的一个链接进去或者跳转过去的。
五：那么客户端的人做什么呢？他们开始封装一些要用的工具类（二维码等等）和集成一些功能（地图、第三支付等等），然后等到原型出来就可以开始做界面了，然后等服务器做一部分接口就开始对接接口填充数据了。
六：做的时候客户端人员需要喝原型设计师沟通，就是给它们开开是不是这个样子？不是就改！如果公司测试模式是“双v”等，则需要边测试边写或者什么的。
七：项目出来了，这时候需要一次集中测试，看有问题没有，没有的话就可以上线了！
八：上线了，不是就没事情做了，这时候会得到客户的一些反馈（如果我不喜欢自己的照片给别人看），那么相关功能的人员就需要更新功能了（这就是迭代吧）。
九：经过一定量的八步骤，可能就需要更新版本了，就是改了再提交审核。

附加：
一：如果你是公司才招进去的，那么你开始的任务是熟悉上一个版本的框架和软件大概（不是让你看懂所有的代码），然后可能需要重新从头写所有的程序（所以没必要把他的思想看很彻底，反正都要自己写），如果不是从头写，那么你就需要好好的看别人的代码了（很痛苦！！！）。
二：你一般是有一个直接上级的，那么你的工作要定期给他讲，这样才能让他了解你，这样才能合理利用资源嘛！
三：如果你是才进入的，那么核心的代码不会让你写的（除非只有你一个程序员），上级会安排一些简单的任务给你，你得高质量完成！等上级给你权利写核心代码的时候，你就已经很厉害了！那么有人就问了，我写简单的对自己没进步啊。那么你就是真的没有工作经验了（工作的人都知道，除了自己特别懒，总感觉时间不够，ps：我都有好多东西想学呢）。
四：代码同步等必备工具，好吧这些你得花两个小时好好看看（只需要两个小时），不要在这上面体现的自己多菜。
五：上级安排你的任务，你不会就谷歌（一定是谷歌），不要总问别人，别人其实都会，你这样去公司是学技术去了？（没人希望新员工是这样，除非BAT等，人家有钱，就爱”培养“）
六：。。。（好吧写不完）

好了，这些就是我的初步理解了（因为我才工作一个月，有错误我后面慢慢改）
